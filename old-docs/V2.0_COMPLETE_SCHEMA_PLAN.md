## Schema Validator Design

### SchemaValidator Class Architecture

```ruby
# lib/uniword/ooxml/schema/schema_validator.rb

module Uniword
  module Ooxml
    module Schema
      # Validates OOXML schema files for correctness and completeness
      #
      # Responsibility: Ensure schema files follow ISO/IEC 29500-1 and
      # internal consistency rules.
      class SchemaValidator
        # Validate all schema files
        #
        # @param loader_config [Hash] Loader configuration
        # @return [ValidationResult] Validation results
        def validate_all(loader_config)
          results = ValidationResult.new

          # Validate loader configuration
          validate_loader_config(loader_config, results)

          # Validate each schema file
          loader_config[:schema_files].each do |schema_file|
            validate_schema_file(schema_file, results)
          end

          # Cross-file validations
          validate_cross_references(results)
          validate_completeness(results)

          results
        end

        private

        # Validate individual schema file
        def validate_schema_file(file_path, results)
          schema = load_schema_file(file_path)

          # Syntax validation
          validate_yaml_syntax(file_path, results)

          # Structure validation
          validate_schema_structure(schema, results)

          # Element validation
          schema[:elements].each do |name, definition|
            validate_element_definition(name, definition, results)
          end
        end

        # Validate element definition
        def validate_element_definition(name, definition, results)
          # Required fields
          validate_required_fields(name, definition, results)

          # Tag format (w:element)
          validate_tag_format(definition[:tag], results)

          # Attributes
          validate_attributes(definition[:attributes], results)

          # Children
          validate_children(definition[:children], results)

          # ISO reference
          validate_iso_reference(definition[:iso_reference], results)
        end

        # Validate cross-file element references
        def validate_cross_references(results)
          # Ensure all child elements are defined
          # Ensure no duplicate definitions
          # Ensure namespace consistency
        end

        # Validate completeness against ISO spec
        def validate_completeness(results)
          # Check coverage percentage
          # Identify missing required elements
          # Suggest priority additions
        end
      end

      # Validation results container
      class ValidationResult
        attr_reader :errors, :warnings, :info

        def initialize
          @errors = []
          @warnings = []
          @info = []
        end

        def add_error(message, file: nil, line: nil)
          @errors << { message: message, file: file, line: line }
        end

        def add_warning(message, file: nil, line: nil)
          @warnings << { message: message, file: file, line: line }
        end

        def add_info(message)
          @info << message
        end

        def valid?
          @errors.empty?
        end

        def summary
          {
            total_errors: @errors.count,
            total_warnings: @warnings.count,
            valid: valid?
          }
        end
      end
    end
  end
end
```

## Migration Guide

### For Existing Code

No breaking changes - existing code continues to work:

```ruby
# Existing code (still works)
schema = Uniword::Ooxml::Schema::OoxmlSchema.load

# New code (optional)
schema = Uniword::Ooxml::Schema::OoxmlSchema.load('config/ooxml/schema_loader.yml')
```

### For Schema Extensions

Use new multi-file approach:

```ruby
# Add new schema file
# config/ooxml/namespaces/main/my_extension.yml

# Add to schema_loader.yml
schema_files:
  - namespaces/main/my_extension.yml
```
