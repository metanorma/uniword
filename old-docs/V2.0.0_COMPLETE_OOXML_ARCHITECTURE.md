### Principle 2: Schema-Driven Serializer

**Single Responsibility**: ElementSerializer only serializes based on schema.

```ruby
module Uniword
  module Ooxml
    # Serializes ANY element based on external OOXML schema definition
    #
    # Responsibility: Serialize elements according to schema rules
    # Single Responsibility - serialization only, schema-driven
    # Open/Closed - new elements via schema, not code changes
    #
    # Architecture:
    # 1. Load OOXML schema from YAML (configuration over convention)
    # 2. For each element, look up schema definition
    # 3. Serialize according to schema rules
    # 4. Recursively serialize children
    class ElementSerializer
      def initialize(schema_file: 'config/ooxml_schema.yml')
        @schema = OoxmlSchema.load(schema_file)
      end

      # Serialize ANY element to OOXML XML
      #
      # @param element [Element] Any Uniword element
      # @param options [Hash] Serialization options
      # @return [String] OOXML XML string
      def serialize(element, options = {})
        # Get schema definition for this element type
        schema_def = @schema.definition_for(element.class)

        # Build XML based on schema
        build_xml_from_schema(element, schema_def, options)
      end

      private

      # Build XML following schema definition
      def build_xml_from_schema(element, schema_def, options)
        builder = Nokogiri::XML::Builder.new(encoding: 'UTF-8') do |xml|
          # Create root element with namespaces from schema
          xml.send(schema_def.tag.to_sym, schema_def.namespaces) do
            # Serialize attributes from schema
            serialize_attributes(xml, element, schema_def)

            # Serialize children from schema
            serialize_children(xml, element, schema_def, options)
          end
        end

        builder.to_xml
      end

      # Serialize attributes according to schema
      def serialize_attributes(xml, element, schema_def)
        schema_def.attributes.each do |attr_def|
          value = element.send(attr_def.property_name)
          next unless value

          xml.parent[attr_def.tag] = format_attribute_value(value, attr_def)
        end
      end

      # Serialize children according to schema
      def serialize_children(xml, element, schema_def, options)
        schema_def.children.each do |child_def|
          child_elements = get_child_elements(element, child_def)

          if child_def.multiple?
            # Serialize each child element
            child_elements.each do |child|
              child_xml = serialize(child, options)
              xml.parent << Nokogiri::XML(child_xml).root
            end
          elsif child_elements
            # Serialize single child
            child_xml = serialize(child_elements, options)
            xml.parent << Nokogiri::XML(child_xml).root
          end
        end
      end
    end
  end
end
```

### Principle 3: Schema Object Model

**MECE**: Schema classes are mutually exclusive, collectively exhaustive.

```ruby
module Uniword
  module Ooxml
    # OOXML Schema object model
    # Loaded from external YAML, represents schema structure
    class OoxmlSchema
      attr_reader :elements

      def self.load(schema_file)
        config = Configuration::ConfigurationLoader.load_file(schema_file)
        new(config)
      end

      def initialize(config)
        @elements = {}
        config[:elements].each do |name, definition|
          @elements[name] = ElementDefinition.new(name, definition)
        end
      end

      # Get schema definition for element class
      def definition_for(element_class)
        element_name = class_to_schema_name(element_class)
        @elements[element_name] || raise("No schema for #{element_class}")
      end

      private

      def class_to_schema_name(klass)
        # Convert Uniword::Paragraph → :paragraph
        klass.name.split('::').last.downcase.to_sym
      end
    end

    # Represents definition of ONE OOXML element type
    # Single Responsibility - hold schema for one element
    class ElementDefinition
      attr_reader :name, :tag, :namespace, :children, :attributes

      def initialize(name, definition)
        @name = name
        @tag = definition[:tag]
        @namespace = definition[:namespace]
        @children = (definition[:children] || []).map { |c| ChildDefinition.new(c) }
        @attributes = (definition[:attributes] || []).map { |a| AttributeDefinition.new(a) }
      end

      def namespaces
        { 'xmlns:w' => @namespace }
      end
    end

    # Represents definition of child element relationship
    class ChildDefinition
      attr_reader :element_type, :tag, :optional, :multiple

      def initialize(definition)
        @element_type = definition[:element]
        @tag = definition[:tag]
        @optional = definition[:optional] != false
        @multiple = definition[:multiple] == true
      end

      def optional?
        @optional
      end

      def multiple?
        @multiple
      end
    end

    # Represents definition of OOXML attribute
    class AttributeDefinition
      attr_reader :name, :tag, :type, :required

      def initialize(definition)
        @name = definition[:name]
        @tag = definition[:tag]
        @type = definition[:type] || :string
        @required = definition[:required] == true
      end

      def property_name
        # Convert OOXML name to Ruby property name
        @name.to_s.gsub(/([A-Z])/, '_\1').downcase.sub(/^_/, '').to_sym
      end
    end
  end
end
```

## Architecture Benefits

### 1. External Configuration (No Hardcoding)

```yaml
# Add new element - NO CODE CHANGES:
# Just edit config/ooxml/schema_main.yml

new_element:
  tag: 'w:newElement'
  namespace: '...'
  children: [...]
  attributes: [...]
```

### 2. Open/Closed Principle

```ruby
# Add new OOXML element:
# 1. Add to YAML schema (open for extension)
# 2. No code changes needed (closed for modification)

# Extend serialization behavior:
# 1. Subclass ElementSerializer
# 2. Register in SerializerRegistry
```

### 3. Single Responsibility

| Class | Single Responsibility |
|-------|---------------------|
| `OoxmlSchema` | Load and hold schema definitions |
| `ElementDefinition` | Define one element's structure |
| `ElementSerializer` | Serialize one element to XML |
| `ElementDeserializer` | Deserialize XML to one element |
| `SchemaValidator` | Validate schema completeness |

### 4. Separation of Concerns

```
Configuration Layer:  OOXML schema YAML files
Schema Layer:        OoxmlSchema, ElementDefinition
Serialization Layer: ElementSerializer, ElementDeserializer
Model Layer:         Document, Paragraph, Run (unchanged)
```

### 5. MECE Organization

**Mutually Exclusive:**
- Each ElementDefinition defines ONE element type
- Each schema file covers ONE OOXML namespace
- No overlapping definitions

**Collectively Exhaustive:**
- All OOXML elements defined in schema
- Fallback to generic serialization if needed
- Complete coverage guaranteed

### Step 2: Implement Schema Classes

```ruby
module Uniword
  module Ooxml
    module Schema
      # Main schema class - loads from YAML
      class OoxmlSchema
        def self.load(schema_file)
          # Delegate to ConfigurationLoader
          config = Configuration::ConfigurationLoader.load_file(schema_file)
          new(config)
        end

        def initialize(config)
          @version = config[:schema][:version]
          @namespace = config[:schema][:namespace]
          @elements = build_element_definitions(config[:elements])
        end

        def definition_for(element_class_or_name)
          name = normalize_name(element_class_or_name)
          @elements[name] || default_definition(name)
        end

        private

        def build_element_definitions(elements_config)
          elements_config.transform_values { |def| ElementDefinition.new(def) }
        end
      end

      # Element definition - ONE element's complete schema
      class ElementDefinition
        attr_reader :tag, :children, :attributes, :namespace

        def initialize(definition)
          @tag = definition[:tag]
          @namespace = definition[:namespace]
          @children = (definition[:children] || []).map { |c| ChildDefinition.new(c) }
          @attributes = (definition[:attributes] || []).map { |a| AttributeDefinition.new(a) }
        end

        # Get namespace declarations for this element
        def namespace_declarations
          # Build from namespace definition
        end

        # Get child definition by name
        def child(name)
          @children.find { |c| c.element_type == name }
        end

        # Get attribute definition by name
        def attribute(name)
          @attributes.find { |a| a.name == name }
        end
      end

      # Child relationship definition
      class ChildDefinition
        attr_reader :element_type, :tag, :optional, :multiple, :required

        def initialize(definition)
          @element_type = definition[:element]
          @tag = definition[:tag]
          @optional = definition.fetch(:optional, true)
          @multiple = definition.fetch(:multiple, false)
          @required = definition.fetch(:required, false)
        end

        def optional?
          @optional
        end

        def multiple?
          @multiple
        end

        def required?
          @required
        end
      end

      # Attribute definition
      class AttributeDefinition
        attr_reader :name, :tag, :type, :required, :values

        def initialize(definition)
          @name = definition[:name]
          @tag = definition[:tag]
          @type = definition.fetch(:type, :string)
          @required = definition.fetch(:required, false)
          @values = definition[:values]  # For enum types
        end

        def property_name
          # Convert w:val → val, w:lineRule → line_rule
          @tag.to_s.sub(/^w:/, '').gsub(/([A-Z])/, '_\1').downcase.to_sym
        end

        def required?
          @required
        end

        def enum?
          !@values.nil?
        end
      end
    end
  end
end
```

### Step 3: Schema-Driven Serialization

```ruby
module Uniword
  module Ooxml
    module Serialization
      # Serializes ANY element using schema definition
      class ElementSerializer
        def initialize(schema: nil)
          @schema = schema || Schema::OoxmlSchema.load('config/ooxml/schema_main.yml')
        end

        # Serialize element to OOXML XML
        def serialize(element, options = {})
          schema_def = @schema.definition_for(element.class)

          Nokogiri::XML::Builder.new(encoding: 'UTF-8') do |xml|
            serialize_element(xml, element, schema_def, options)
          end.to_xml
        end

        private

        def serialize_element(xml, element, schema_def, options, parent = nil)
          # Build element with schema-defined tag
          xml.send(schema_def.tag.to_sym, schema_def.namespace_declarations) do
            # Serialize attributes according to schema
            schema_def.attributes.each do |attr_def|
              serialize_attribute(xml, element, attr_def)
            end

            # Serialize children according to schema
            schema_def.children.each do |child_def|
              serialize_child(xml, element, child_def, options)
            end
          end
        end

        def serialize_attribute(xml, element, attr_def)
          value = element.send(attr_def.property_name)
          return unless value || attr_def.required?

          # Format value according to type
          formatted = format_value(value, attr_def.type)
          xml.parent[attr_def.tag] = formatted
        end

        def serialize_child(xml, element, child_def, options)
          child_elements = get_children(element, child_def)

          if child_def.multiple?
            child_elements.each do |child|
              serialize_element(xml, child, child_def.schema, options)
            end
          elsif child_elements
            serialize_element(xml, child_elements, child_def.schema, options)
          elsif child_def.required?
            # Serialize empty required element
            xml.send(child_def.tag.to_sym)
          end
        end
      end
    end
  end
end
```
